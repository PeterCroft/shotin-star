function drawConfetti(){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  let alive=false;
  pieces.forEach(p=>{
    p.y+=p.dy;p.x+=p.dx;p.r+=p.dr;
    if(p.y>canvas.height*.6)p.alpha=Math.max(0,p.alpha-.02);
    if(p.alpha>0)alive=true;
    ctx.save();ctx.globalAlpha=p.alpha;ctx.translate(p.x,p.y);ctx.rotate(p.r);ctx.fillStyle=p.color;ctx.fillRect(-p.w/2,-p.h/2,p.w,p.h);ctx.restore();
  });
  if(alive){raf=requestAnimationFrame(drawConfetti)}else{canvas.style.display='none';ctx.clearRect(0,0,canvas.width,canvas.height)}
}
